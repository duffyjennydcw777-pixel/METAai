def get_telegram_creds():
    env_path = Path(__file__).parent.parent / ".env"
    
    if not env_path.exists():
        logging.warning(f"Env file not found: {env_path}")
        return None, None

    try:
        from dotenv import load_dotenv
        load_dotenv(env_path)
    except ImportError:
        logging.error("python-dotenv not installed")
        return None, None

    token = os.environ.get(TELEGRAM_BOT_TOKEN_ENV)
    chat_id = os.environ.get(TELEGRAM_CHAT_ID_ENV)

    if not token or not chat_id:
        logging.error("Incomplete Telegram credentials")
        return None, None

    return token, chat_id
