```python
# Пример безопасного формирования payload
def create_secure_payload(plan_id: str, user_id: int, payment_id: int) -> str:
    import secrets
    import hashlib
    
    salt = secrets.token_hex(16)
    payload = f"{plan_id}:{user_id}:{payment_id}:{salt}"
    signature = hashlib.sha256(payload.encode()).hexdigest()
    
    return f"{payload}:{signature}"
```
