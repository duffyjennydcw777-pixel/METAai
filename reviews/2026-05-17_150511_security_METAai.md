```python
def safe_subprocess_run(cmd, cwd=None):
    try:
        result = subprocess.run(
            cmd, 
            capture_output=True, 
            text=True, 
            timeout=10,
            cwd=cwd,
            encoding="utf-8", 
            errors="strict",  # Строгая обработка ошибок кодировки
            check=True  # Вызывать исключение при ошибке
        )
        return result.stdout
    except (subprocess.CalledProcessError, UnicodeError) as e:
        logging.error(f"Ошибка выполнения команды: {e}")
        raise
```

```python
def validate_path(path):
    """Безопасная проверка пути"""
    abs_path = os.path.abspath(path)
    
    # Проверка, что путь находится в допустимой директории
    allowed_roots = [
        os.path.abspath(Path.home()),
        os.path.abspath(Path(__file__).parent.parent)
    ]
    
    if not any(abs_path.startswith(root) for root in allowed_roots):
        raise ValueError(f"Недопустимый путь: {abs_path}")
    
    # Проверка прав доступа
    if not os.access(abs_path, os.R_OK):
        raise PermissionError(f"Нет прав чтения: {abs_path}")
    
    return abs_path
```

```python
def safe_regex_search(pattern, text, flags=0):
    """Безопасный поиск с экранированием"""
    try:
        safe_pattern = re.escape(pattern)
        return re.search(safe_pattern, text, flags)
    except re.error as e:
        logging.error(f"Ошибка регулярного выражения: {e}")
        return None
```
